### 사용자 정의 이터러블을 통해 알아보기

```javascript
const iterable = {
  [Symbol.iterator]() {
    let i = 3;
    return {
      next() {
        return i === 0 ? { done: true } : { value: i--, done: false };
      },
    };
  },
};

let iterator = iterable[Symbol.iterator]();
log(iterator.next());
log(iterator.next());
log(iterator.next());
log(iterator.next()); // done: true
```

# 웰폼드 이터레이터

```javascript
const arr2 = [1, 2, 3];
let iter2 = arr2[Symbol.iterator](); // 실행 결과 자기 자신을 반환함, 웰폼드 이터레이터, 웰폼드 이터러블 이라고 한다.
log(iter2[Symbol.iterator]);
iter2.next();
for (const a of iter2) log(a); // 2, 3
```

# 웰폼드 이터레이터 제작

```javascript
const iterable = {
  [Symbol.iterator]() {
    let i = 3;
    return {
      next() {
        return i === 0 ? { done: true } : { value: i--, done: false };
      },
      [Symbol.iterator]() {
        return this; //웰폼드 이터레이터
      },
    };
  },
};

for (const a of iterator) {
  log(a);
} //정상 순회

let iterator = iterable[Symbol.iterator]();
log(iterator.next()); // 3

for (const a of iterator) log(a); // 2, 1
```
